💻 Async Pattern

async loadUsers() {
  this.isLoading.set(true);
  this.loadError.set(null);

  try {
    const users = await firstValueFrom(
      this.http.get<User[]>('/api/users')
    );
    this.users.set(users);
  } catch (error) {
    this.loadError.set((error as Error).message);
  } finally {
    this.isLoading.set(false);
  }
}
